Key architectural decision (from 18-RESEARCH.md):

- mcp-server.mjs runs as a SEPARATE OS PROCESS from daemon/server.mjs
- Each process may safely open one kuzu.Database (single-writer constraint is per-process)
- mcp-server.mjs is read-only — only runs SELECT queries, no writes
- Concurrent reads are safe with Kuzu WAL mode

```javascript
server.registerTool(
  'get_related',
  {
    description:
      'Get documents related to a given document ID via Kuzu graph traversal. Supports reverse traversal to find documents that reference this one.',
    inputSchema: {
      doc_id: z.number().int().describe('Document ID to traverse from'),
      hops: z.number().int().min(1).max(3).default(2).describe('Maximum traversal depth (1-3)'),
      direction: z
        .enum(['forward', 'reverse', 'both'])
        .default('forward')
        .describe('Traversal direction: forward (outgoing), reverse (incoming), both'),
    },
  },
  async ({ doc_id, hops, direction }) => {
    try {
      const results = await kuzuFindRelated(kuzuDb, doc_id, hops, direction);
      return {
        content: [
          {
            type: 'text',
            text: JSON.stringify(
              { doc_id, hops, total: results.length, related: results },
              null,
              2
            ),
          },
        ],
      };
    } catch (err) {
      return {
        content: [{ type: 'text', text: JSON.stringify({ error: err.message }) }],
        isError: true,
      };
    }
  }
);
```
