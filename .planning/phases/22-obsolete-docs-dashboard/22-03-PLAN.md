### JavaScript (in a `<script>` tag, NO inline handlers)

```javascript

var allRows = [];
var sortKey = 'confidence_score';
var sortDir = -1; // -1 = DESC

function escapeHtml(str) {
  if (!str) return '';
  return String(str)
    .replace(/&/g, '&amp;')
    .replace(/</g, '&lt;')
    .replace(/>/g, '&gt;')
    .replace(/"/g, '&quot;');
}

function showError(msg) {
  var el = document.getElementById('error-banner');
  el.textContent = msg;
  el.style.display = 'block';
}

function hideError() {
  document.getElementById('error-banner').style.display = 'none';
}

function confColor(score) {
  if (score >= 0.8) return '#dc2626';
  if (score >= 0.7) return '#d97706';
  if (score >= 0.5) return '#16a34a';
  return '#6366f1';
}

function renderTable(rows) {
  var tbody = document.getElementById('table-body');
  var table = document.getElementById('signals-table');
  var empty = document.getElementById('empty-state');
  if (rows.length === 0) {
    table.style.display = 'none';
    empty.style.display = 'block';
    return;
  }
  empty.style.display = 'none';
  table.style.display = 'table';
  tbody.innerHTML = rows.map(function(row) {
    var conf = (row.confidence_score * 100).toFixed(0) + '%';
    var barColor = confColor(row.confidence_score);
    var flagClass = 'badge badge-' + escapeHtml(row.flag_label).replace('-', '-');
    return '<tr data-id="' + row.id + '">'

      + '<td><input type="checkbox" class="row-check" data-id="' + row.id + '"></td>'

      + '<td>'

      + '<span class="conf-bar"><span class="conf-fill" style="width:' + (row.confidence_score * 100) + '%;background:' + barColor + '"></span></span>'

      + escapeHtml(conf)

      + '</td>'

      + '<td><span class="' + flagClass + '">' + escapeHtml(row.flag_label) + '</span></td>'

      + '<td>' + escapeHtml(String(row.age_days)) + '</td>'

      + '<td>' + escapeHtml(row.repository) + '</td>'

      + '<td title="' + escapeHtml(row.path) + '">' + escapeHtml(row.filename) + '</td>'

      + '<td>' + escapeHtml(String(row.inbound_link_count)) + '</td>'

      + '<td><button class="btn btn-sm btn-dismiss" data-id="' + row.id + '">Dismiss</button></td>'

      + '</tr>';

  }).join('');
  updateSelectedCount();
}

function applyFilters() {
  var repo = document.getElementById('filter-repo').value;
  var flag = document.getElementById('filter-flag').value;
  var text = document.getElementById('filter-text').value.toLowerCase();
  var sorted = allRows.slice().sort(function(a, b) {
    var av = a[sortKey], bv = b[sortKey];
    if (av == null) av = '';
    if (bv == null) bv = '';
    if (av < bv) return -1 * sortDir;
    if (av > bv) return 1 * sortDir;
    return 0;
  });
  var filtered = sorted.filter(function(r) {
    if (repo && r.repository !== repo) return false;
    if (flag && r.flag_label !== flag) return false;
    if (text && r.path.toLowerCase().indexOf(text) === -1) return false;
    return true;
  });
  renderTable(filtered);
}

function updateSelectedCount() {
  var checked = document.querySelectorAll('input.row-check:checked').length;
  document.getElementById('selected-count').textContent = checked + ' selected';
  document.getElementById('btn-archive').disabled = checked === 0;
}

function getSelectedIds() {
  return Array.from(document.querySelectorAll('input.row-check:checked'))
    .map(function(cb) { return Number(cb.dataset.id); });
}

function populateRepoFilter(rows) {
  var repos = Array.from(new Set(rows.map(function(r) { return r.repository; }))).sort();
  var sel = document.getElementById('filter-repo');
  repos.forEach(function(repo) {
    var opt = document.createElement('option');
    opt.value = escapeHtml(repo);
    opt.textContent = repo;
    sel.appendChild(opt);
  });
}

function updateStats(data) {
  var el = document.getElementById('stats-summary');
  el.textContent = data.total + ' signal(s) total — showing ' + data.count + ' active';
}

async function loadData() {
  document.getElementById('loading').style.display = 'block';
  document.getElementById('signals-table').style.display = 'none';
  document.getElementById('empty-state').style.display = 'none';
  hideError();
  try {
    var res = await fetch('/obsolete?limit=500');
    if (!res.ok) throw new Error('Server error: ' + res.status);
    var data = await res.json();
    allRows = data.rows || [];
    populateRepoFilter(allRows);
    updateStats(data);
    applyFilters();
  } catch (err) {
    showError('Failed to load data: ' + err.message);
  } finally {
    document.getElementById('loading').style.display = 'none';
  }
}

// Event listeners — NO inline onclick handlers (helmet blocks unsafe-inline)
document.addEventListener('DOMContentLoaded', function() {
  loadData();

  document.getElementById('btn-refresh').addEventListener('click', loadData);

  document.getElementById('btn-reset').addEventListener('click', function() {
    document.getElementById('filter-repo').value = '';
    document.getElementById('filter-flag').value = '';
    document.getElementById('filter-text').value = '';
    applyFilters();
  });

  document.getElementById('filter-repo').addEventListener('change', applyFilters);
  document.getElementById('filter-flag').addEventListener('change', applyFilters);
  document.getElementById('filter-text').addEventListener('input', applyFilters);

  document.getElementById('select-all').addEventListener('change', function() {
    document.querySelectorAll('input.row-check').forEach(function(cb) {
      cb.checked = document.getElementById('select-all').checked;
    });
    updateSelectedCount();
  });

  document.getElementById('table-body').addEventListener('change', function(e) {
    if (e.target.classList.contains('row-check')) updateSelectedCount();
  });

  // Sort by column header click
  document.querySelectorAll('th[data-sort]').forEach(function(th) {
    th.addEventListener('click', function() {
      var key = th.dataset.sort;
      if (sortKey === key) {
        sortDir *= -1;
      } else {
        sortKey = key;
        sortDir = -1;
      }
      document.querySelectorAll('th[data-sort]').forEach(function(h) {
        h.classList.remove('sort-asc', 'sort-desc');
      });
      th.classList.add(sortDir === -1 ? 'sort-desc' : 'sort-asc');
      applyFilters();
    });
  });

  // Per-row Dismiss button (event delegation)
  document.getElementById('table-body').addEventListener('click', async function(e) {
    if (!e.target.classList.contains('btn-dismiss')) return;
    var id = Number(e.target.dataset.id);
    try {
      var res = await fetch('/obsolete/' + id + '/dismiss', { method: 'POST' });
      if (!res.ok) throw new Error('Dismiss failed: ' + res.status);
      // Remove row from allRows and re-render
      allRows = allRows.filter(function(r) { return r.id !== id; });
      applyFilters();
    } catch (err) {
      showError('Dismiss failed: ' + err.message);
    }
  });

  // Archive Selected (batch dismiss)
  document.getElementById('btn-archive').addEventListener('click', async function() {
    var ids = getSelectedIds();
    if (ids.length === 0) return;
    if (!confirm('Dismiss ' + ids.length + ' document(s) from the dashboard for 30 days?\n\nNo documents will be deleted.')) return;
    try {
      var res = await fetch('/obsolete/batch-dismiss', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ ids: ids })
      });
      if (!res.ok) throw new Error('Batch dismiss failed: ' + res.status);
      var idSet = new Set(ids);
      allRows = allRows.filter(function(r) { return !idSet.has(r.id); });
      document.getElementById('select-all').checked = false;
      applyFilters();
    } catch (err) {
      showError('Archive failed: ' + err.message);
    }
  });
});

```
