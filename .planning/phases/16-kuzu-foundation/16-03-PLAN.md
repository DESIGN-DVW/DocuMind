Replace the existing `/health` handler body (currently lines 65-72):

```javascript

app.get('/health', async (_req, res) => {
  try {
    db.prepare('SELECT 1').get();             // SQLite probe (existing)
    const conn = new kuzu.Connection(kuzuDb); // Kuzu probe
    const result = await conn.query('RETURN 1');
    await result.getAll();
    try { conn.close(); } catch (_) {}
    res.json({
      status: 'ok',
      version: '2.0.0',
      uptime: process.uptime(),
      mcp_mode: MCP_MODE,
      kuzu: { status: 'ok', path: KUZU_DIR },
    });
  } catch (err) {
    res.status(503).json({ status: 'error', error: err.message });
  }
});

```

### Change 4 — Close kuzuDb in shutdown() before db.close()

In the `shutdown()` function, insert `kuzuDb.close()` (or best-effort equivalent per 16-01-SUMMARY.md) immediately BEFORE `db.pragma('wal_checkpoint(TRUNCATE)')`:

```javascript

function shutdown(signal) {
  console.error(`[DocuMind] ${signal} received — shutting down gracefully`);
  server.close(() => {
    try {
      try { kuzuDb.close(); } catch (_) { /* Kuzu GC handles if close() unavailable */ }
      db.pragma('wal_checkpoint(TRUNCATE)');
      db.close();
    } catch (_err) {
      // Ignore errors during cleanup
    }
    process.exit(0);
  });
  setTimeout(() => process.exit(1), 5000).unref();
}

```
