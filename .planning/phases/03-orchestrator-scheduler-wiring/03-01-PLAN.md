  ```javascript

  const siblingsByDir = new Map();
  for (const doc of docs) {
    const dir = path.dirname(doc.path);

    siblingsByDir.get(dir).push(doc);
  }

  ```

### **2. Replace the existing sibling block (section 4, lines 110-128) with:**

  ```javascript

  // 4. Same-folder siblings = related_to (capped)
  const dirPath = path.dirname(doc.path);
  const siblingsInDir = siblingsByDir.get(dirPath) || [];
  // Skip bulk directories (dispatch dirs with 50+ files)
  if (siblingsInDir.length <= 50) {
    const siblings = siblingsInDir
      .filter(d => d.id !== doc.id && d.id > doc.id)
      .slice(0, 10); // cap: max 10 sibling edges per doc
    for (const sibling of siblings) {
      insertRel.run(
        doc.id, sibling.id, 'related_to', 0.3,
        JSON.stringify({ reason: 'same_folder', auto_detected: true }),
        now
      );
      edgeCount++;
    }
  }

  ```
