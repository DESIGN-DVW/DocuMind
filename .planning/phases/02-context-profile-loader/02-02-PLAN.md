```javascript
function commonDir(paths) {
  if (!paths.length) return '';
  const parts = paths[0].split('/');
  let common = '';
  for (let i = 0; i < parts.length; i++) {
    const candidate = parts.slice(0, i + 1).join('/');
    if (paths.every(p => p.startsWith(candidate + '/'))) common = candidate;
    else break;
  }
  return common;
}
